0.1 mysql(installer install)
	utf8, utf8_general_ci

0.2 apache (installer install)
	then copy use ipv6 binary for ipv6 support
	
	add:
	#start
		<Directory ~ "\.svn">
		Order allow,deny
		Deny from all
		</Directory>
	#end
	
	to apache httpd.conf to deny access to .svn folder
	
	modify 
		AllowOverride None in <Directory "D:/apache/2.2.9/htdocs"> ... </Directory>
	to  
		AllowOverride All
	for .htaccess permission control
	
0.3 php (binary install)
	add php folder to system environment path and php
	turn off magic_quotes_gpc, why? see ./the_hash_problem.txt
	set short_open_tag = On
	set ext folder
	turn on extension=php_mysql.dll for mysql functions
	turn on extension=php_mbstring.dll for mb_string functions
	comment out output_buffering = 4096 for debugging
	
	add:
	#start
		Loadfile "D:/php/5.2.6/php5ts.dll"
		LoadModule php5_module "D:/php/5.2.6/php5apache2_2.dll"
	
		PHPIniDir "D:/php/5.2.6"
	
		AddHandler application/x-httpd-php .php
		AddHandler application/x-httpd-php-source .phps	
	#end
	to apache httpd.conf
